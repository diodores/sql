create table if not exists singer (id serial primary key, name varchar(100) not null);
create table if not exists album (id serial primary key, title varchar(70) not null,
album_release_year integer);
create table if not exists track (id serial primary key, track_title varchar(40) not null unique, duration time not null, album_id integer references singer(id));
create table if not exists genre (id serial primary key, genre_name varchar(70) not null unique);
